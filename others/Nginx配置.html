# Nginx基础配置与操作

  • nginx常用命令
  • nginx.conf 基础配置

配置文件在conf目录下，执行指令为sbin目录下的nginx脚本文件。

# nginx常用命令

进入sbin目录下进行如下操作指令，此时前面的./nginx就是sbin下的nginx脚本文件：

./nginx # 启动nginx，默认conf/nginx.conf文件作为配置文件

./nginx -c conf_file # 以配置文件conf_file的配置内容启动，conf_file为配置文件的绝对路径

./nginx -t # 检测默认配置文件是否有语法错误

./nginx -t -c conf_file # 检测配置文件conf_file是否有语法错误，conf_file为配置文件的绝对路径

./nginx -s reload # 平滑的重启，在修改完配置后重新加载配置文件时非常有用

./nginx -s stop # 直接关闭停止nginx

./nginx -s quit # 有序退出停止，其会等所有请求结束后关闭nginx，比直接stop更平滑

ps -ef|grep nginx # 查看nginx的进程号，其中输出的master为主进程

# user                         nobody staff;
worker_processes             2;

events {
  worker_connections       1024;
}

# http 协议的配置
http {
  include               mime.types; # 这里是引入了外部单独的 mime.types 文件的内容插入该位置（设置了types字段对应相关文件的mime类型）
  default_type          text/html; # 默认文件类型
  client_max_body_size  50m;   # 客户端最大提交内容大小，主要影响文件上传
  gzip                  on;  # 开启 gzip 静态资源压缩，下面 gzip_types 为要压缩的文件 mime 类型
  gzip_types            text/css text/x-component application/x-javascript application/javascript text/javascript text/x-js text/richtext image/svg+xml text/plain text/xsd text/xsl text/xml image/x-icon;

  sendfile              on;

  # 服务端口监听，可以设置多个监听不同端口
  server {
    listen               8888 default_server; # 监听端口，default_server 表示此端口设为默认端口
    server_name  127.0.0.1;   # 拦截的访问IP，也可设为 localhost

    root                 /Users/computerName/Desktop/demo; # 该端口指向的根目录位置

    # 路由匹配，可设置多个
    location /path { # 这里注意'/path'既能匹配'/path/x'，也能匹配'/pathx'，而'/path/'只能匹配'/path/x'
      # proxy_set_header Host $host:9999; # 设置响应头部？
      proxy_pass https://finance.pae.baidu.com/; # 将匹配路径代理到的实际路径，可以是在线地址，也可以是本地文件/服务地址，proxy_pass地址后面带上'/'则会减去location匹配的部分进行拼接，否则会直接拼接在后面
    }

    # 不设置则默认全量？若有设置部分不在里面的文件类型可能加载不出来
    location ~* \.(js|css|flash|media|jpg|png|gif|dll|svg|cab|CAB|ico|woff|woff2|ttf|json|mp4|mp3|zip)$ {
      root   /root_path;
      index  index.html index.htm;
      expires  30d; # 失效时长？
    }

    # vue/react history 路由模式配置示例
    location /busiMonitorSys {
      root /Users/computerName/Desktop/demo/dist;
      index    index.html index.htm;  # 当访问到一个目录时，默认将其指向其中的哪个实际文件（按顺序查找列举的可能文件）
      try_files $uri $uri/ /dist/index.html;  # 关键配置，使得支持history模式
      # 关键字 $uri 表示访问时匹配到的完整路径
      # try_files 这里表示若有找到 $uri 指向的文件则返回对应文件内容，否则返回 $uri/ 下上面 index 字段设置的可能文件，若还是未找到，则返回最后设置的文件内容
      # 服务器将vue/react history路由按常规文件路由的方式查找一般是找不到对应文件的，所以匹配到try_files这里最后指定的文件
      # 这里让其返回前端工程打包后的入口文件index.html的内容(实际路径根据自己的部署位置修改)，注意返回的是内容而不是重定向浏览器中的访问路径，故内容正确且路由无异常就可使得JS中的router按照history模式正确解析
    }
  }

# 代码检查与格式化

  • Prettier (opens new window)：代码格式化
  • ESLint (opens new window)：JS 代码检查
  • Stylelint (opens new window)：CSS 代码检查
  • lint-staged (opens new window)：git 暂存文件检查（根据配置自动修复prettier/eslint/stylelint的可修复内容）
  • husky (opens new window)：git 钩子
  • commitlint (opens new window)：git 提交信息检查
  • commitizen (opens new window)：git 提交信息规范化交互式命令行工具（可选）

相关依赖包可参考touch-vue3-template中的package.json。

上面lint-staged字段配置的是在执行git commit时，对暂存区各类文件进行的对应格式化或检查。

scripts/prepare字段的脚本会在工程npm install安装依赖时自动执行，这里是husky的命令，用于初始化 git hooks，载入其依赖的相关文件，若要手动初始化则执行npx husky init。

.husky/pre-commit 文件内容：

npx lint-staged

.husky/commit-msg 文件内容：

npx --no -- commitlint --edit $1

# Prettier

在与ESLint同时使用时可安装：

  • eslint-config-prettier：配置，用于关闭与 Prettier 冲突的 ESLint 规则
  • eslint-plugin-prettier：插件，用于使用 Prettier 来修复 ESLint 中格式相关的错误（冲突时就会报错）
npm i -D eslint-config-prettier eslint-plugin-prettier

注意：在使用eslint-plugin-prettier/recommended即推荐配置时其会同时自动配置eslint-config-prettier，否则就需要自行分别引入并放入配置项（但包安装需要手动安装，不会预安装）

在 .eslintrc.js（v8）中配置：

{
  "extends": ["plugin:prettier/recommended"]
}

在 eslint.config.js（v9）中配置：

import eslintPluginPrettierRecommended from 'eslint-plugin-prettier/recommended';

export default [
  /* 其他配置 */
  eslintPluginPrettierRecommended,
]

以下配置基于 prettier v3+ 版本。

// prettier.config.js
export default {
  printWidth: 100, // 单行字符推荐长度（非ESLint中max-len最大长度限制）
  tabWidth: 2,  // 缩进空格数
  useTabs: false, // 使用tab缩进（这里我们使用空格来缩进）
  semi: true, // 所有句尾添加分号
  singleQuote: true,  // 使用单引号
  quoteProps: 'as-needed',  // 对象属性名称不加引号（仅在需要加时才加），若要想vue中class的对象属性又都加上引号，目前在 prettier 下该配置项就只能使用 preserve 值，即保留用户输入，所有地方由用户自己决定，而不统一
  jsxSingleQuote: false,  // jsx中不强制使用单引号，因为部分类HTML的内容属性上是要双引号的，而其他JS部分的内容还是推荐单引号
  trailingComma: 'es5',  // 尾逗号（仅在多行对象、数组等场景下，TS下包括函数参数列表，方便尾部追加选项）
  bracketSpacing: true,  // 在对象内部两端保留空格
  bracketSameLine: false,  // 多行HTML属性时标签的尾部尖括号不与最后一个属性放在同一行（默认true，即放在同一行，不适用于自闭和标签）
  // jsxBracketSameLine: false,  // 同 bracketSameLine，只是针对 jsx //!! v2.4.0 中已废弃，由 bracketSameLine 代替
  arrowParens: 'avoid',  // 箭头函数参数只有一个时是否添加括号（avoid，尽可能省略）
  // requirePragma: true,  // 是否文件开头有 @prettier 或 @format 的注释该文件才被格式化 //! 在历史大项目中可以开启，以便逐步或针对性格式化，避免一次格式化影响大量文件
  // insertPragma: false,  // 是否向已格式化的文件顶部插入 @format 注释标记，默认 false
  htmlWhitespaceSensitivity: 'css',  // 标签内两端空格敏感度，css：根据CSS的默认display属性来决定是否保留空格，行内元素不保留，块级元素保留，而其他类HTML则保留用户输入的空格
  vueIndentScriptAndStyle: false,  // vue中的style和script标签是否缩进
  endOfLine: 'lf',  // 换行风格（lf（仅换行符\n，即Linux/Unix风格）、crlf（回车符+换行符\r\n，即Windows风格）
  embeddedLanguageFormatting: 'auto',  // 是否格式化嵌入的代码块，比如JS中的html模板、md中的代码块（默认auto，即自动根据内容识别来格式化，off为关闭） //!! 实测关闭后会影响.vue文件正常内容的格式化
  singleAttributePerLine: false,  // 是否强制 HTML 及 vue、jsx 等类 html 标签的每个属性都单独一行（我们更倾向属性较多需要换行时才单独成行）
}

# ESLint

//!! 局部忽略/启用 eslint 检查的注释语法，[xxx] 为可选的具体规则名(多个规则名用逗号隔开)，不指定则表示禁用所有规则（单行注释多行注释格式都可以）

/* eslint-disable [xxx] */  //! 忽略当前行之后的所有内容 eslint 检查，直到遇到 eslint-enable 注释行
/* eslint-enable [xxx] */  //! 重新启用当前行之后的所有内容 eslint 检查
/* eslint-disable-line [xxx] */  //! 忽略当前行 eslint 检查
/* eslint-disable-next-line [xxx] */  //! 忽略该注释下一行 eslint 检查

注意下面配置文件中 extends 继承的配置需要安装相关的包（注意 ESLint 版本不同可能对应的包也有所不同）。

npm i -D eslint-config-prettier

# 原常规配置

此为 v8- 版本中的配置方式。

// .eslintrc.js
module.exports = {
  root: true,
  extends: [
    'eslint:recommended', // eslint 官方推荐风格
    'plugin:vue/vue3-recommended', // vue 官方推荐风格，由 eslint-plugin-vue 提供
    '@vue/eslint-config-prettier/skip-formatting', // create-vue 脚手架配置，目的跳过格式化内容
    'plugin:prettier/recommended', // 使用 prettier 风格覆盖 ESLint 中与之冲突的规则
    './.eslintrc-auto-import.json', // 当使用了 unplugin-auto-import 插件时，需要将其配置的 api 配置为全局变量，避免 eslint 报错（该文件需要在配置时开启 eslintrc.enabled 选项来生成，之后再关闭，避免每次都去生成）
  ],
  parserOptions: {
    ecmaVersion: 'latest',
  },
  rules: rules
};

# 新flat扁平化配置

此为 v9+ 版本中的配置方式（v8.57+也支持）。

// eslint.config.js
import eslint from "@eslint/js";
import pluginVue from 'eslint-plugin-vue';
import eslintConfigPrettier from 'eslint-config-prettier';
import globals from 'globals';
import skipFormatting from '@vue/eslint-config-prettier/skip-formatting';
import autoImport from './.eslintrc-auto-import.json';
//!! 若在vue中使用自动导入插件来省去 vue/pinia/element-plus 等的 import 语句时，还需要配在该配置文件中将这些api配置为global全局变量

export default [
  //! 单独使用一个配置对象作为全局配置对象
  {
    name: 'app/files-to-lint', // name 字段的主要作用是抛出异常时定位是哪个规则触发的，可选
    files: ['**/*.{js,mjs,jsx,vue}'],
  },
  {
    name: 'app/files-to-ignore',
    // '**/node_modules/' 和 '.git/' 是默认忽略的
    ignores: ['**/dist/**', '**/dist-ssr/**', '**/coverage/**', 'public/'], // 忽略指定目录下的文件 eslint 检查，v9+ 版本中已没提到使用单独的忽略配置文件，故最好使用该配置项统一指定
  },
  {
    languageOptions: {
      globals: {
        ...globals.browser,
        ...autoImport.globals,
      },
    },
  },
  eslint.configs.recommended, // eslint 官方推荐风格，//! 若值关注下面自定义rules中的配置可以移除这里的推荐配置，避免一些不想限制的规则默认开启
  skipFormatting,
  eslintConfigPrettier, // 使用prettier配置风格，故这里关闭所有与prettier格式化冲突的eslint风格选项
  ...pluginVue.configs['flat/recommended'], // 使用 vue 官网推荐的风格(vue3)，这里用得到是最全的推荐（包含 Priority A/B/C）
  {
    // 放到对象内 extends 字段中主要场景为单独针对该对象内 files 字段匹配的文件使用，且需要搭配 plugins 字段注册，放在外部则针对前面全局匹配的所有文件
    // extends: [
    //   eslint.configs.recommended,
    //   eslintConfigPrettier,
    //   ...pluginVue.configs['flat/recommended'],
    // ],
    rules: rules // 自定义的规则，覆盖前面的规则
  }
]

# 部分 rules 说明

const rules = {
  //!! 与 prettier 完全冲突的格式规则项（与 prettier 同时使用时要关闭，可使用 eslint-config-prettier 或 eslint-plugin-prettier 来统一关闭）
  "semi": [2, "always"], // 强制使用分号结尾，警告级别：忽略（0==="off"）、警告（1==="warn"）、报错（2==="error"）
  "comma-style": [2, "last"], // 逗号放句末而非句首
  "comma-dangle": [2, "always-multiline"], // 多行且最后一行与结尾的]或}不同行时尾随逗号(若为 only-multiline 则不强制)，否则不可尾随逗号
  "nonblock-statement-body-position": [2, "beside"], // 强制要求单行语句不换行
  "brace-style": [2, "1tbs", { "allowSingleLine": true }], // 采用一个块的开头大括号与它对应的语句或声明放在同一行中的风格，且允许单行代码块
  "space-before-blocks": [2, "always"], // 强制在块之前使用一致的空格
  "keyword-spacing": [2, { "before": true, "after": true }], // 强制在关键字前后使用一致的空格
  "switch-colon-spacing": ["error", {"after": true, "before": false}], // 强制在 switch 中 case 和 default 的冒号后使用空格
  "space-in-parens": [2, "never"], // 禁止或强制圆括号内两侧没有空格
  "array-bracket-spacing": [2, "never"], // 强制数组方括号内两侧没有空格
  "object-curly-spacing": [2, "always"], // 强制对象大括号内两侧有空格
  "space-infix-ops": [2, { "int32Hint": true }], // 强制操作符周围有空格（除 num|0 形式表示32位有符号整数）
  "key-spacing": [2, { "beforeColon": false, "afterColon": true, "mode": "strict" }], // 强制在对象字面量属性中键和值之间使用一致的空格
  "arrow-spacing": [2, { "before": true, "after": true }], // 强制箭头函数箭头前后使用一致的空格
  "space-before-function-paren": [2, { "anonymous": "never", "named": "never", "asyncArrow": "always" }], // 禁止或强制在 function 的左括号之前使用空格（异步箭头函数除外）
  "eol-last": [2, "always"], // 强制在文件末尾使用换行符（即保留至少一个空行）
  "padded-blocks": [2, "never"], // 禁止在块语句、函数体、类静态块、类和 switch 语句的内部开头和结尾出现空行
  "quote-props": [2, "as-needed", { "number": true }], // 强制对象字面量的属性只在必须加引号的情况下才加，否则就不加（数字属性需要加上）//? 但在vue的模板中使用对象添加class时又希望统一加或不加（consistent-as-needed），可看下针对vue的配置风格
  "arrow-parens": [1, "as-needed", { "requireForBlockBody": true }], // 推荐箭头函数参数只有一个参数且使用的return简写语法时省略参数小括号包裹
  "wrap-iife": [2, "any", { "functionPrototypeMethods": true }], // 强制立即执行函数表达式要用小括号包裹，即 (function(){}()) 或 (function(){})()，包括使用 .call 和 .apply 调用的函数表达式

  //! 与 prettier 存在部分冲突的格式规则项（可自行根据情况在配置文件中启用，在 eslint-config-prettier 或 eslint-plugin-prettier 中也是默认关闭的）
  "curly": [2, "multi-line", "consistent"], // 强制代码块使用一致的大括号风格（允许单行，同语句下要么都大括号要么都单行）
  "max-len": [2, {
    "code": 100,  // 强制一行文本长度最大不超过100个字符
    "tabWidth": 2,
    "ignoreComments": true, // 忽略注释
    "ignoreTrailingComments": true, // 忽略尾部注释
    "ignoreUrls": true, // 忽略URL
    "ignoreStrings": true, // 忽略包含引号字符串的行
    "ignoreTemplateLiterals": true, // 忽略字符串模板
    "ignoreRegExpLiterals": true // 忽略正则
  }],
  "no-mixed-operators": 1, // 混合使用不同的操作符而未用括号分组时将警告（除了同优先级的操作符以及四则运算符）
  "quotes": [2, "single"], // 强制优先使用单引号
  "prefer-arrow-callback": 1, // 推荐使用箭头函数作为回调函数 //!! 与 prettier 可能存在冲突bug，当自动修复后反而有bug时可尝试关闭
  "arrow-body-style": [1, "as-needed"], // 推荐可省略大括号的地方就省略 //!! 与 prettier 可能存在冲突bug，当自动修复后反而有bug时可尝试关闭

  //! 与 prettier 不存在冲突的格式规则项
  "no-empty": [2, { "allowEmptyCatch": true }], // 除 catch 外不允许空代码块（可至少加一行注释）
  "max-lines": [2, 1000], // 强制文件最大行数不超过1000行
  "max-lines-per-function": [2, 80], // 强制函数最大行数不超过80行（指整个函数而非只是函数体）

  //! 语言特性
  "no-var": 2, // 要求使用 let 或 const 而不是 var
  "no-undef": 2, // 禁止未声明的变量，除非它们在 /global/ 下定义
  "prefer-const": [2, { "destructuring": "all", "ignoreReadBeforeAssign": true }], // 要求优先使用 const 声明那些声明后不再被修改的变量
  "one-var": [1, "never"], // 建议每个变量分开声明（大多情况下分开合适，但像 for 循环条件中的变量，不建议分开）
  "no-unused-vars": [2, {  // 禁止出现未使用过的变量
    "vars": "all",
    "args": "after-used", // 忽略函数最后使用的形参之前未使用的形参
    "caughtErrors": "none", // 忽略 catch 中的形参
    "ignoreRestSiblings": true  // 忽略解构中使用了扩展运算符的变量
  }],
  "no-use-before-define": [2, { "functions": false }], // 禁止在变量/类声明前使用它们（函数例外），包括在函数中访问的情况（因若访问时机在声明前仍会出错）
  "no-shadow": [1, { "builtinGlobals": true, allow: ['name', 'event'] }], // 警告变量声明与外层作用域的变量同名（包括函数参数、JS内置全局变量），但允许指定的部分（因name较常用，按规则却冲突了window.name）
  "no-redeclare": [2, { "builtinGlobals": true }], // 禁止在同层作用域下重复声明变量或函数（包括函数参数和JS内置全局变量，主要针对 var 声明的情况）
  "no-multi-assign": [2, { "ignoreNonDeclaration": true }], // 禁止连续赋值，除非都是已声明的变量（即不含任何声明语句）
  "no-new-wrappers": 2, // 禁止对 String、Number 和 Boolean 使用 new 操作符（即不允许 new String('foo')）
  "radix": 1, // 推荐在 parseInt() 使用基数参数（即第二个参数）
  "no-extra-boolean-cast": 1, // 警告不必要的布尔转换（即 if (foo) {} 而不是 if (!!foo) {}）
  "prefer-template": 1, // 推荐使用模板字符串而非字符串连接
  "no-useless-escape": 2, // 禁止出现无意义的转义字符
  "no-array-constructor": 2, // 禁止使用 Array 构造函数（即 new Array()），除非初始只是创建一个指定长度的空数组
  "array-callback-return": [2, { "allowImplicit": true }], // 强制数组的某些回调函数中 return 语句的存在
  "no-new-object": 2, // 禁止使用 Object() 构造函数（即 new Object()）
  "object-shorthand": [2, "always", { "avoidQuotes": true }], // 强制对象字面量属性/方法名称使用简写形式（但若属性/方法名包含引号则优先使用完整形式，不含箭头函数）
  "dot-notation": 2, // 强制尽可能地使用点号取值（即不推荐 foo['bar'] 的形式）
  "prefer-destructuring": 1, // 推荐优先使用数组和对象解构而非索引访问
  "no-prototype-builtins": 1, // 推荐禁止直接在对象上调用 Object.prototype 的方法
  "no-new-func": 2, // 禁止使用 Function 构造函数（即 new Function()）
  "no-inner-declarations": 2, // 禁止在嵌套的非函数块中声明函数（但函数表达式形式是可以的）
  "prefer-rest-params": 2, // 使用剩余参数代替 arguments
  "no-param-reassign": [2, { "props": true }], // 禁止对函数的形参重新赋值或修改子属性

# Stylelint

局部忽略/启用 stylelint 检查的注释语法，[xxx] 为可选的具体规则名(多个规则名用逗号隔开)，不指定则表示禁用所有规则，注意CSS中只有/**/注释语法，SCSS中额外扩展了//注释语法。

/* stylelint-disable [xxx] */  //! 忽略当前行之后的所有内容 stylelint 检查，直到遇到 stylelint-enable 注释行
/* stylelint-enable [xxx] */  //! 重新启用当前行之后的所有内容 stylelint 检查
/* stylelint-disable-line [xxx] */  //! 忽略当前行 stylelint 检查
/* stylelint-disable-next-line [xxx] */  //! 忽略该注释下一行 stylelint 检查

依赖包安装：

npm i -D stylelint stylelint-scss stylelint-order postcss-html postcss-scss

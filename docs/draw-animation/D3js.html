# 零散API

# data.join()

const svg = d3.select('body').append('svg');
svg.selectAll("circle")
  .data(dataList)
  .join("circle")   // join()这一句是 v5+ 中新增的对 enter、update、exit 方法的合并简写，
  .attr("fill", "none")
  .attr("stroke", "black");

// 上面 join 方法当只传入一个字符串时，就等价于下面的写法
// join 方法可传入三个参数，分别是 enter、update、exit 三种情况对应的操作函数(参数为对应的元素集)
svg.join(
  enter => enter.append("circle"),  // enter 为数据长度多出选中的元素集长度的部分，这里操作就是根据多出的数据部分插入相应的元素
  update => update, // 默认，update 为选中元素长度与数据长度相同的部分，默认这部分元素会根据数据更新
  exit => exit.remove() // 默认，exit 为选中元素长度多于数据长度的部分，默认多处的元素会被删除
);

// 上面 join 传入一个字符串的情况用 v4- 的方法可改写如下
svg.enter().append("circle");

# selection.call()

svg.append('g').call(fn, args...); // call() 方法中传入一个函数，这里的作用就是将当前元素集合作为参数放进`fn`函数中进行处理，后面的都是额外传递给函数的参数
// 上面的方法可以看作是如下效果
fn(svg.append('g'), args...);

# 计算Math

// 常量，注意都是大写
Math.PI;                // 返回圆周率 π 的值
Math.E;                 // 返回算数常量自然对数的底数 e 的值
Math.LN2;               // 返回 2 的自然对数
Math.LN10;              // 返回 10 的自然对数
Math.LOG2E;             // 返回以 2 为底数的 e 的对数
Math.LOG10E;            // 返回以 10 为底数的 e 的对数
Math.SQRT2;             // 返回 2 的平方根
Math.SQRT1_2;           // 返回 2 的平方根的倒数，即 2 的负二分之一次方，注意1与2之间是下划线而不是中划线

// 取值
Math.random();          // 返回一个 0~1 之间的随机数，能取到 0，但取不到 1
Math.abs(x);            // 返回 x 的绝对值
Math.min(...x);         // 返回传入参数中的最小值，可任意多个参数
Math.max(...x);         // 返回传入参数中的最大值，可任意多个参数

// 取舍
Math.ceil(x);           // 对 x 小数部分只进不舍
Math.floor(x);          // 对 x 小数部分只舍不进
Math.round(x);          // 对 x 小数部分四舍五入

// 指数、对数
Math.exp(n);            // 返回常量 e 的 n 次幂
Math.sqrt(x);           // 返回 x 的平方根，即二分之一次幂
Math.pow(x, n);         // 返回 x 的 n 次幂
Math.log(x);            // 返回以 e 为底数的 x 的对数

// 三角函数
Math.sin(rad);          // 返回 rad 弧度角的正弦值，注意是弧度
Math.cos(rad);          // 返回 rad 弧度角的余弦值，注意是弧度
Math.tan(rad);          // 返回 rad 弧度角的正切值，注意是弧度
Math.asin(x);           // 返回 x 的反正弦值，即由正弦值得到弧度值
Math.acos(x);           // 返回 x 的反余弦值，即由余弦值得到弧度值
Math.atan(x);           // 返回 x 的反正切值，即由正切值得到弧度值
Math.atan2(y, x);       // 返回(x, y)坐标点与原点连线与X轴正方向形成的弧度角值(-π ~ π 之间)，注意参数是(y, x)，横纵坐标位置相反

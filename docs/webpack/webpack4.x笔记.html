    # 初始化

    # 工程目录下初始化一个package.json文件，"private":true设置项表示私有项目
npm init

# 项目内安装webpack(最新版)及脚手架，npx可限定在当前目录下运行指令(区分全局安装与局部安装的包)
npm install webpack webpack-cli -D

    # package.json文件

    以下紧包含package.json文件中部分配置项。json文件中不能像js文件添加注释，以下注释仅用于解释说明使用，实际使用应移除。

    {
  "name": "notebook",
  "version": "1.0.0",
  "main": "index.js",
  "license": "MIT", //MIT表示开源协议
  "scripts": {
    "dev": "webpack-dev-server", //开发环境运行
    "start": "npm run dev",
    "build": "webpack" //打包运行（生产环境）
  }
}

    # webpack.config.js配置文件

    const path = require('path');
const HtmlWebpackPlugin = require('html-webpack-plugin'); //npm install html-webpack-plugin -D，生成html文件并插入打包文件
const CleanWebpackPlugin = require('clean-webpack-plugin'); //npm install clean-webpack-plugin -D，打包时先清除之前的打包内容
const webpack = require('webpack'); //引入webpack，以手动开启其内置的部分功能，如热更新

let babelLoader;
let styleLoader;
let fileLoader;

module.exports = {
  mode: 'production', //环境模式，可省略，默认为’production’，还有’development’、’test’等
  //入口文件，默认情况下打包输出文件名字使用对应的键名key命名
  //entry: './index.js', //单个入口文件，同entry: {main: ‘./index.js’}
  entry: { //多入口文件
    main: './index.js',
    app: './app.js'
  },
  output: { //打包输出文件名及位置等
    // publicPath: 'http://cdn.com.cn', //若打包的静态资源要放到cdn或其他位置，我们就可以设置该项用于在插入的文件前加上该域名或地址(webpack会自动在所有打包资源的引用前加上该路径)
    //filename: 'bundle.js', //指定为具体文件名，若不指定文件名则默认为‘main.js’，该配置方式只能用于单入口文件
    filename: '[name]-[hash:7].js', //使用[placeholder]占位符命名打包输出文件名，name为对应入口文件的key，hash为根据打包文件内容生成的哈希值，:7表示使用7位哈希值
    path: path.resolve(__dirname, 'dist') //可省略，默认就是dist目录下
  },
  devtool: 'cheap-module-eval-source-map', //设置使用哪种方式来映射打包文件与源文件间的关系(开发环境下debug很有用)，生产环境模式下可用‘cheap-module-source-map’或关闭(none，默认)
  //借助‘webpack-dev-server’开启一个web服务器并自动打开浏览器并监听文件变动后自行进行编译或打包等处理再自动刷新浏览器
  //注意：若webpack未内置‘webpack-dev-server’则需要自己安装：npm install webpack-dev-server -D
  devServer: { //默认将在8080端口建立服务
    contentBase: './dist', //服务器应该起在哪个文件夹下，即打包后资源根目录
    open: true, //run dev的时候自动在打开浏览器中打开服务地址
    port: 8089, //将访问端口设置为8089，默认8080
    proxy: {
      '/api': 'http://localhost:3000', //服务代理，当请求访问‘/api’时将自动转发到‘http://localhost:3000’
    },
    hot: true //启用热更新(hmr)，即只更新修改了的内容，而不是对整个页面刷新（避免一些需要手动触发出现的内容一刷新就又要手动去触发才能看到）
    // hotOnly: true //即使热更新开启了但若出现未生效就不自动刷新页面(默认webpack-dev-server检测到文件更新后重新处理后会自动刷新浏览器页面)
  },
  /**================================= 模块处理，即配置loader，告诉webpack如何处理各种文件 =================================*/
  module: {
    rules: [    //模块请求规则数组
      //脚本文件处理
      babelLoader,
      //样式文件处理
      ...styleLoader,
      //文件处理(非文本文件，如图片、音视频、字体等等)
      ...fileLoader
    ],
    // noParse 配置项让 Webpack 忽略对部分没采用模块化的文件的递归解析和处理，这样做的好处是能提高构建性能。
    // 忽略的文件中不应含有 import, require, define 等模块化的调用，不然会导致构建出的代码中包含无法在浏览器环境下执行的模块化语句
    // 例如一些库jQuery、lodash 它们庞大又没有采用模块化标准，若让 Webpack 去解析这些文件耗时又没有意义
    noParse: /jquery|lodash/   // 值类型：RegExp | [RegExp] | function，这里使用正则表达式
    // noParse: function(content) {    // 使用函数(可以是使用箭头函数)，Webpack 3+ 支持函数形式，content 代表一个模块的文件路径
    //   return /jquery|lodash/.test(content);    // 返回 true or false，为true的文件则将被排除
    // }
  },
  /**==================================================== 插件配置 ====================================================*/
  //插件可以在webpack运行到某个时刻的时候，帮我们做一些事(类似vue组件中在某个生命周期钩子中做一些特定的处理)
  plugins: [
    //html-webpack-plugin插件用于在打包结束后自动生成一个html文件(或指定一个自定义的模板html文件)，并把打包生成的js文件自动引入到这个html文件中
    //注意：若未指定自定义的模板html文件，而是自动生成html文件，则该html内只包含文档的基本结构(文档声明、head、body，没有其他DOM)
    new HtmlWebpackPlugin({
      template: 'src/index.html' //使用自定义的html文件模板
    }),
    //clean-webpack-plugin插件用于在打包前先清除原先的打包输出文件
    new CleanWebpackPlugin(),
    //实例化热更新插件，webpack-dev-server中内置插件，配置后上面devServer中的hot:true才有效
    new webpack.HotModuleReplacementPlugin()
  ]
}

    # 模块内容处理-loader

    # ES6语法转换相关loader：babel-loader

    babelLoader = {
  test: /\.js$/,    // 使用这个正则匹配文件类型，可以一个正则去匹配多种文件，也可以使用数组设置多个正则，每个正则间是 或 关系
  // 排除不使用该loader进行处理的一些位置
  // exclude: /node_modules/, // 因模块安装的内容一般都已转成ES5，又因安装的包一般都比较多，若不排除该部分每次编译将很费时间，若有未转的模块，则可以使用下面的include选项单独列出
  include: path.resolve(__dirname, 'src'),    // 限定规则应用的文件位置，默认所有，这里只匹配src目录内容，多个可使用数组
  // 对匹配文件应用的loader列表，处理顺序为从后到前，？后面是传给紧邻loader的参数，另 loaders 是 use 的别名
  // 且字符串值是loader的简写方式：use:[ "style-loader” ] 是 use:[ { loader: "style-loader "} ]的简写
  use: [
    /** babel转换ES6语法，相关配置结合webpack和babel官网查看 */
    // 'babel-loader?cacheDirectory',   // 这里cacheDirectory参数用于缓存 babel 编译结果加快重新编译速度
    // 若该loader需要传递多个参数或其他定制化配置，则可以写成下面Object形式
    { // babel-loader用于引导处理一些ES6+语法，以便转换为主流浏览器都支持的ES5语法，注意其只是转移一些新的语法，并不能转移新的api，新api只能安装并引入`@babel/polyfill`兼容
      loader: 'babel-loader',
      enforce:'post',    // enforce:'post' 表示把该 loader 的执行顺序放到最后，若值为pre，则是放到最前
      // babel相关的配置项可能较多，其提供另一种方式来单独设置下面的options部分，可以在项目根目录下创建一个`.babelrc`的文件保存options设置（文件中不能有注释）
      options: {
        // presets: ['@babel/preset-env'], // @babel/preset-env 模块是实际用于将ES6+转为ES5语法的模块，其包含了相关翻译转换规则（需安装）
        presets: [['@babel/preset-env', {
          // 第二个参数更多设置项可参考webpack官方文档
          // 下面这个设置的意思是根据用到的一些ES6+中的新api来按需引入对应的polyfill，这样安装并`import @babel/polyfill`时就只会引入用到的polyfill(可减少未用到的polyfill部分的文件体积)
          useBuiltIns: 'usage'
        }]],
        cacheDirectory:true, // 缓存 babel 编译结果加快重新编译速度
        // 上面配置presets转义ES6语法和引入polyfill的方式适用于一般项目开发，而对于开发一个插件或是UI组件库，上面引入的polyfill都是全局变量方式，会造成全局变量污染
        // 这时应该采用下面plugins插件的方式配置（需安装），该方式会采用闭包的方式去转译和引入相关polyfill
        plugins: [['@babel/plugin-transform-runtime', {
          'corejs': 2, // 注意这里需要安装`@babel/runtime-corejs2`
          'helpers': true,
          'regenerator': true,
          'useESModules': false
        }]]
      }
    }
  ],
  // 解析器精准配置
  // 因 Webpack 是以模块化的 JS 文件为入口，所以内置了对模块化 JS 的解析功能，支持 AMD、CommonJS、SystemJS、ES6。
  // parser 属性可以更细粒度的配置哪些模块语法要解析，哪些不解析，和 noParse 配置项的区别在于 parser 可以精确到语法层面， 而 noParse 只能控制哪些文件不被解析
  parser: {    // 这些选项默认未设置值，为undefined，与true一样表示默认开启使用
    amd: false, // 禁用 AMD
    commonjs: false, // 禁用 CommonJS
    system: false, // 禁用 SystemJS
    harmony: false, // 禁用 ES6 import/export
    requireInclude: false, // 禁用 require.include
    requireEnsure: false, // 禁用 require.ensure
    requireContext: false, // 禁用 require.context
    browserify: false, // 禁用 browserify
    requireJs: false, // 禁用 requirejs
    node: false,     // 禁用 __dirname, __filename, module, require.extensions, require.main 等
    node: {/*...*/}     // 在模块级别(module level)上重新配置 node 层(layer)
  }
}

    babel相关的配置项可能较多，其提供另一种方式来单独设置下面的options部分，可以在项目根目录下创建一个.babelrc的文件保存options设置（类似一个json文件，注意文件中也不能有注释）

    # 样式相关loader

    styleLoader = [
  {
    test: /\.css$/,
    use: ['style-loader', 'css-loader'],    // 先给 css-loader 处理，再将结果给 style-loader 处理。css-loader用于分析样式文件间的引用(@import)关系，最后将引用文件合并到一个样式文件中；style-loader用于将合并的样式挂载到HTML文件head中(开发模式下将以style标签的形式将样式内联到html页面中)，注意同一文件用到多个loader时，loader的处理顺序是从后往前的
    exclude: path.resolve(__dirname, 'node_modules')    // 不应用该规则的文件位置，与include刚好相反，多个可使用数组
  },
  {
    test: /\.scss$/,
    //use: ['style-loader', 'css-loader’, ’sass-loader’, ‘postcss-loader'], //scss文件需要使用sass-loader，否则webpack不认识scss文件；使用postcss-loader的目的是使用其自动添加样式浏览器前缀的插件(具体查看官网配置方法，需要增加一个postcss.config.js文件配置引用‘autoprefixer’插件)
    use: [
      'style-loader',
      {
        loader: 'css-loader',
        options: {
          importLoaders: 2 //js等文件中import scss文件时会应用所有loader处理，但是若是在scss文件内@import的其他样式文件可能不会走下面的两个loader处理而直接从css-loader开始处理，这里设置importLoaders意思就是在样式文件中使用import引入的样式文件也要走下面的两个loader先进行处理
        }
      },
      'sass-loader',
      'postcss-loader'
    ],
    exclude: path.resolve(__dirname, 'node_modules')
  }
]

    # 文件(非文本文件)处理相关loader

    fileLoader = [
  // 文件处理(非文本文件，如图片、音视频、字体等等)
  {
    test: /\.(gif|png|jpe?g|eot|woff|ttf|svg|pdf)$/, // 对非文本文件采用 file-loader 加载
    //loader: 'file-loader',    // loader 是 use: [ { loader } ] 的简写，只有一个loader的时候简洁写法
    use: {
      loader: 'file-loader',
      options: { //单独配置项，若不需要修改可以不配置直接使用默认
        name: '[name]_[hash:7].[ext]' //[xxx]是placeholder占位符，表示特定含义的内容，’:7’表示取值长度为七个字符，所有占位符可在loader配置文档中查看，注意最外层的引号，默认是[hash].[ext]
      }
    }
  },
  //url-loader的作用与file-loader类似，只是url-loader处理的文件在大小小于限定的字节数时将进行base64编码并返回编码后的DataURL替换原本的文件使用，大于的就同file-loader一样（一般来说推荐这样使用，如图片小于几KB时就采用base64格式减少请求）
  {
    test: /\.(gif|png|jpe?g)$/,
    use: {
      loader: 'url-loader',
      options: {
        limit: 1024*7 //7KB以内的图片将编码成base64格式，大于的就同file-loader
      }
    }
  }
]

# HTML标签用法

# 普通链接

<!-- 链接网址 -->
<a href="http://www.baidu.com">target指定打开位置，默认当前窗口打开，"_blank"在新窗口打开，"iframeName"表示在名为"iframeName"的iframe中打开</a>

# 在线打开文件

<!-- Chrome中可以直接打开图片、PDF、MP3音频、MP4视频查看 -->
<a href="url/image.jpg" target="_blank">图片</a>
<a href="url/报告.pdf">pdf文件</a>

# 下载文件

<!-- 默认只要是浏览器不能直接打开预览的文件都会进行下载 -->
<a href="url/file.rar">rar压缩文件</a>
<a href="url/file.ppt">ppt文件</a>
<a href="url/file.xlsx">excel表格</a>
<a href="url/file.docx">word文档</a>

<!-- 添加 download 属性，避免浏览器可能直接在线打开，只有 Firefox 和 chrome 支持 -->
<a href="url/image.jpg" download="下载文件名" title="download中下载文件名可不设置，但download属性要保留">图片下载</a>

# 发邮件

点击链接时会跟根据href中的邮件信息调用设备的邮件应用并填充好相关内容。

  • mailto:指定邮件给某人或某些人(多个地址逗号隔开)，其余的参数都放在问号?之后，多参数则用&连接(跟一般url参数类似)
  • cc参数为抄送地址
  • bcc参数为加密抄送地址
  • subject参数为邮件的主题
  • body参数为邮件的正文内容
<!-- 邮件给某人，多人则邮箱地址用逗号隔开 -->
<a href="mailto:xx@xx.com">mailto:邮箱</a>
<!-- 加抄送 -->
<a href="mailto:xx@xx.com?cc=xxx@xx.com">cc=抄送邮箱</a>
<!-- 加密抄送 -->
<a href="mailto:xx@xx.com?bcc=xxx@xx.com">bcc=加密抄送邮箱</a>
<!-- 加主题 -->
<a href="mailto:xx@xx.com?subject=主题">subject=主题</a>
<!-- 加邮件正文 -->
<a href="mailto:xx@xx.com?body=正文">body=正文内容</a>
<!-- 邮件给多人 -->
<a href="mailto:xx@xx.com,xxx@xx.com">mailto:邮箱1,邮箱2</a>

<!-- 完整实例 -->
<a href="mailto:a@xx.com,b@xx.com?cc=c@xx.com&bcc=d@yy.com&subject=主题&body=正文">发邮件</a>

# 拨打电话

移动端时点击链接时会根据href中的电话信息调用设备的拨号应用并填充好相关内容，通过tel:xxx指定要拨打的电话号码。

<!-- Web页面一键拨号 -->
<a href="tel:10086">tel:电话号码</a>
<!-- 使用 wtai 协议进行拨号 -->
<a href="wtai://wp/mc;10086">wtai:协议地址;电话号码"</a>

# 发短信

移动端时点击链接时会根据href中的短信信息调用设备的短信应用并填充好相关内容。**注意：**发短信的时候有可能会出现乱码，这个与页面的编码格式有关系，需要将页面的编码格式改为 Unicode(UTF-8)。

  • sms:指定发短信给某人或某些人(多个号码逗号隔开)，其余的参数放在问号?之后(跟一般url参数类似)
  • body参数为短信内容
<!-- 发短信给某人，多人则电话号码用逗号隔开 -->
<a href="sms:10086">sms:电话号码</a>
<!-- 发送已编辑好的短信内容 -->
<a href="sms:10086?body=内容">body=短信内容</a>
<!-- 给多个号码发送短信 -->
<a href="sms:10086,10010?body=内容">多个号码间用逗号隔开</a>

# JS中自动解析URL

使用<a>标签自动解析URL，返回类似window.location包含信息的对象。

function parseURL(url){
  const a = document.createElement('a');
  a.href = url;
  return {
    source: decodeURIComponent(url),   //decodeURIComponent方法将URL中编码过的字符进行解码(如汉字等)
    protocol: a.protocol.replace(':',''),   // protocol属性返回当前 URL 的协议（前面的http、https等，原本包含:号）
    host: a.hostname,   // hostname属性返回当前 URL 的主机名
    port: a.port||'80', // port属性返回当前 URL 的端口号
    path: decodeURIComponent(a.pathname), // 返回当前 URL 的路径部分（即主机名后的文件路径）
    file: (decodeURIComponent(a.pathname).match(/\/([^\/?#]+)$/i) || [,''])[1], //当前页文件名
    hash: decodeURIComponent(a.hash).replace('#',''),   // hash属性返回从井号 (#) 开始的 URL（锚）(原本包含#号)
    query: decodeURIComponent(a.search).replace('?',''),     // search属性返回从问号 (?) 开始的 URL（查询部分）(原本包含?号)
    params: (function(){    //查询字段与相应值解析
      var ret = {},
        seg = decodeURIComponent(a.search).replace('?', '').split('&'),
        s;
      for (var len=seg.length,i=0;i<len;i++) {
        if (!seg[i]) continue;
        s = seg[i].split('=');
        ret[s[0]] = s[1];
      }
      return ret;
    })()
  }
}
const aLink = parseURL('https://wanje.github.io/html/a标签的各种用途.html?id=xy&pwd=123456#ada');

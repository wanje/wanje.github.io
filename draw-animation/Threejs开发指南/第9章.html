# 9.1.2 选择对象

这里选择对象也就是第一章中提到的处理鼠标事件判断鼠标是否选中相应对象的问题，这里采用第一章中屏幕二维坐标转换为三维坐标的方法加发射光线来检测选中的物体：

  • 首先，基于屏幕上的点击位置(归一化)创建一个THREE.Vector3向量；
  • 接着，使用vector.unproject方法将屏幕上的归一化的点击坐标转换为三维场景中的坐标；
  • 然后，使用THREE.Raycaster创建一束从摄像机位置到场景中鼠标点击方向的投射光线；
  • 最后，使用raycaster.intersectObjects方法判断指定对象集合中哪些对象被该光线照射到了，而返回的第一个照射到的对象就是点击到的最前面的网格对象
function handleClick(event) {
  // 归一化画布坐标
  const x = (event.offsetX / renderer.domElement.clientWidth) * 2 - 1;
  const y = -(event.offsetY / renderer.domElement.clientHeight) * 2 + 1;
  const vector = new THREE.Vector3(x, y, 0.5);
  vector.unproject(camera); // 将归一化坐标投射到三维空间坐标中

  // 创建光线投射，第一个参数表示光线发射起点位置、第二个参数为一个方向向量表示光线的发射方向
  const raycaster = new THREE.Raycaster(camera.positon, vector.sub(camera.positon).normalize());
  // 检测光线穿过那些物体，第一个参数为要检测的对象集合，可选第二个布尔参数表示是否检测物体的后代对象(默认false)，返回一个数组
  const intersectObjects = raycaster.intersectObjects(scene.children);

  // 判断点击到的物体是否是我们的目标物体
  if (intersectObjects.length > 0) {
    if (intersectObjects[0].object.name === 'targetName') {
      console.log(`intersectObjects数组中的每个网格对象都包含如下一些信息，其中
        ‘distance’表示摄像机到被点击对象的具体，
        ‘face’和‘faceIndex’表示该网格中被选中的面及该面在所有面中的索引位置，
        ‘point’表示网格上那个点被点击了，
        ‘uv’表示点击位置所对应的2D纹理的uv值，
        ‘object’则为该点中网格的mesh对象
      `);
      console.log('点击到了目标物体', intersectObjects[0].object);
    }
  }
}
